150 /*
151  * don't play with this unless you benchmark!
152  * decompression is not dependent on the hash function
153  * the hashing function might seem strange, just believe me
154  * it works ;)
155  */
156 #ifndef FRST
157 # define FRST(p) (((p[0]) << 8) | p[1])
158 # define NEXT(v,p) (((v) << 8) | p[2])
159 # if ULTRA_FAST
160 # define IDX(h) ((( h >> (3*8 - HLOG)) - h ) & (HSIZE - 1))
161 # elif VERY_FAST
162 # define IDX(h) ((( h >> (3*8 - HLOG)) - h*5) & (HSIZE - 1))
163 # else
164 # define IDX(h) ((((h ^ (h << 5)) >> (3*8 - HLOG)) - h*5) & (HSIZE - 1))
165 # endif
166 #endif
167 /*
168  * IDX works because it is very similar to a multiplicative hash, e.g.
169  * ((h * 57321 >> (3*8 - HLOG)) & (HSIZE - 1))
170  * the latter is also quite fast on newer CPUs, and compresses similarly.
171  *
172  * the next one is also quite good, albeit slow ;)
173  * (int)(cos(h & 0xffffff) * 1e6)

214 /*
215  * Decompress data compressed with some version of the lzf_compress
216  * function and stored at location in_data and length in_len. The result
217  * will be stored at out_data up to a maximum of out_len characters.
218  *
219  * If the output buffer is not large enough to hold the decompressed
220  * data, a 0 is returned and errno is set to E2BIG. Otherwise the number
221  * of decompressed bytes (i.e. the original length of the data) is
222  * returned.
223  *
224  * If an error in the compressed data is detected, a zero is returned and
225  * errno is set to EINVAL.
226  *
227  * This function is very fast, about as fast as a copying loop.
228  */
